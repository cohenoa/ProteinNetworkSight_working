Automatic Deployment(work in progress):
    Production:
        ./deploy-prod.bat

    Beta:
        echo "DEPLOY_MODE=beta" > .env
        ./deploy-beta.bat


Manual Deployment:
    Production:
        docker compose -p medmolnet-prod -f docker-compose.yaml up -d
    
    Beta:
        docker compose -p medmolnet-beta -f docker-compose.yaml -f docker-compose.beta.yaml up -d
